# Usage Examples

```tsx
import Calendar from "your-calendar-package";

export default function Demo() {
  return (
    <div>
      <h2>Default Calendar</h2>
      <Calendar onChange={(date) => console.log(date)} />

      <h2>Jalali Calendar, RTL</h2>
      <Calendar CalendarType="jalali" dir="rtl" />

      <h2>Custom Day Render</h2>
      <Calendar
        renderDay={(day, date, isCurrentMonth) => (
          <span style={{ color: isCurrentMonth ? "blue" : "gray" }}>{day}</span>
        )}
      />
    </div>
  );
}
```
